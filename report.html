Data Preparation

targets::tar_load(dt_patients)
targets::tar_load(dt_encounters)
targets::tar_load(dt_payer_transitions)

# Calculate age from birthdate
dt_patients[, age := as.numeric(difftime(Sys.Date(), as.Date(birthdate), units = "days")) / 365.25]

# Count number of encounters per patient (healthcare utilization)
encounter_counts <- dt_encounters[, .(n_encounters = .N), by = patient]

# Join patients with encounter counts and coverage
analysis <- dt_patients[encounter_counts, on = .(id = patient), nomatch = 0]
analysis <- analysis[, .(id, age, gender, race, n_encounters, healthcare_coverage)]

# Remove deceased patients
analysis <- analysis[is.na(dt_patients[match(analysis$id, dt_patients$id), deathdate]) |
                     dt_patients[match(analysis$id, dt_patients$id), deathdate] == ""]
